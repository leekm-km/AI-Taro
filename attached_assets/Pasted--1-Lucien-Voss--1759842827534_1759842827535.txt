   - 한국어 ko, 영어 en, 중국어 zh, 태국어 th

3) API 라우트 수정 (예: src/pages/api/tarot.ts)
-------------------------------------------------
import type { NextApiRequest, NextApiResponse } from "next";
import OpenAI from "openai";
import { PERSONAS } from "@/data/personas";

export default async function handler(req: NextApiRequest, res: NextApiResponse) {
  if (req.method !== "POST") return res.status(405).end();
  const { character, language, category, question } = req.body || {};
  const persona = PERSONAS.find(p => p.id === character);
  if (!persona) return res.status(400).json({ error: "invalid_persona" });

  const sys = [
    `당신은 타로 리더 AI다. 캐릭터: ${persona.nameKo} (${persona.nameEn}).`,
    `말투 지침: ${persona.styleGuide}`,
    `출력 언어: ${language} (요청 언어로만 답변).`,
    `점사 주제: ${category}.`,
    `카드 구성: 3장 가정(메이저/마이너 임의). 각 카드 의미→종합 조언.`,
    `금지: 의료/법률 확정 단정, 과도한 운세 단언. 조심스러운 표현.`,
  ].join("\n");

  const user = [
    `질문: ${question || "구체 질문 없음"}`,
    `상황에 맞게 ${category} 관점에 초점.`,
  ].join("\n");

  const client = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });
  const completion = await client.chat.completions.create({
    model: "gpt-4o-mini",
    messages: [{ role: "system", content: sys }, { role: "user", content: user }],
    temperature: 0.7,
  });

  const reading = completion.choices[0]?.message?.content || "";
  res.status(200).json({ reading });
}
